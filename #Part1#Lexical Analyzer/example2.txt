class Message{
   string text;
   Message next;

   Message(){
     this.text = null;
     this.next = null;
   }
 }

class MessageStore{
  Message first;

  MessageStore(){
    this.first = null;
  }

  void append(Message m){
    m.next = first;
    first = m;
  }
}
